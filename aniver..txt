<script>
  // Floating hearts
  const emojis = ['💕','💗','💓','💞','🌸','✨','🎀','💝'];
  const bg = document.getElementById('bgH');
  for(let i=0;i<22;i++){
    const el=document.createElement('span');
    el.className='bg-heart';
    el.textContent=emojis[Math.floor(Math.random()*emojis.length)];
    el.style.cssText=`left:${Math.random()*100}vw;font-size:${14+Math.random()*26}px;animation-duration:${8+Math.random()*14}s;animation-delay:${Math.random()*12}s`;
    bg.appendChild(el);
  }

  // Carousel
  let cur=0; const N=4;
  const slides=()=>document.querySelectorAll('.slide');
  const dots=()=>document.querySelectorAll('.dot');
  function goTo(n,dir){
    const sl=slides(),dt=dots();
    const d=dir!==undefined?dir:(n>cur?1:-1);
    sl[cur].classList.remove('active'); sl[cur].classList.add(d>0?'go-left':'go-right');
    setTimeout(()=>{sl[cur].classList.remove('go-left','go-right');},450);
    cur=((n%N)+N)%N;
    sl[cur].classList.add('active');
    dt.forEach((d2,i)=>d2.classList.toggle('active',i===cur));
  }
  function move(d){goTo(cur+d,d);}

  // Love letter popup
  function openP(){document.getElementById('ov').classList.add('open');document.body.style.overflow='hidden';}
  function closeP(){document.getElementById('ov').classList.remove('open');document.body.style.overflow='';}
  function bgClose(e){if(e.target===document.getElementById('ov')) closeP();}
  document.addEventListener('keydown',e=>{if(e.key==='Escape') closeP();});
</script>
